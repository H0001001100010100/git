//git基本操作

git init
git add *
git commit -m "说明"
git status
git reset --hard HEAD~num|HEAD^
git log
git log --pretty=oneline
git checkout -- 文件 
git reflog  //操作记录
git reset HEAD 文件
git diff HEAD -- 文件  //对比两个文件不同



//git 分支

git branch //获取分支
git checkout -b dev  //创建并切换dev分支
git branch 分支
git checkout 分支  //切换分支
git merge 分支   //合并分支
git branch -d 分支   //删除分支

git merge --no-ff   //禁止快速合并，快速合并查看不到合并记录

git stash    //保存现场
git stash pop  //恢复现场，用于bug修复



//git 远程管理

git clone "网址"   //克隆
git push origin 分支名称   //推送一个分支到远程
git branch --set-upstream-to=origin/远程分支 本地分支   //本地分支跟踪远程分支
git pull origin 分支名称   //拉去远程分支
